SELECT
    co.OrderNumber,
    i.ItemName AS ProductName,
    coi.Quantity AS ProductQuantity,
    SUM(coi.Quantity * si.Quantity * r.Price) AS MaterialsCostForOrderItem
FROM dbo.CustomerOrders co
INNER JOIN dbo.CustomerOrderItems coi
    ON coi.OrderID = co.OrderID
INNER JOIN dbo.Items i
    ON i.ItemID = coi.ItemID
INNER JOIN dbo.Specifications s
    ON s.ItemID = i.ItemID
INNER JOIN dbo.SpecificationItems si
    ON si.SpecificationID = s.SpecificationID
INNER JOIN dbo.Resources r
    ON r.ResourceID = si.ResourceID
GROUP BY
    co.OrderNumber,
    i.ItemName,
    coi.Quantity
ORDER BY
    co.OrderNumber,
    i.ItemName;
