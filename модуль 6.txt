using System;
using System.Net.Http;
using System.Text.RegularExpressions;
using System.Windows.Forms;

namespace WindowsFormsApp3
{
    public partial class Form1 : Form
    {
        private readonly HttpClient httpClient = new HttpClient();
        private string currentPhone = "";

        public Form1()
        {
            InitializeComponent();
        }

        private async void btnGetData_Click(object sender, EventArgs e)
        {
            try
            {
                string url = "http://127.0.0.1:4444/TransferSimulator/mobilePhone";
                string json = await httpClient.GetStringAsync(url);

                currentPhone = ExtractValue(json);
                lblPhoneValue.Text = currentPhone;
                lblResultValue.Text = "";
            }
            catch (Exception ex)
            {
                MessageBox.Show("Ошибка при получении данных: " + ex.Message);
            }
        }

        private void btnCheckResult_Click(object sender, EventArgs e)
        {
            try
            {
                if (string.IsNullOrWhiteSpace(currentPhone))
                {
                    MessageBox.Show("Сначала получите данные");
                    return;
                }

                bool criterion1 = currentPhone.StartsWith("+7");
                bool criterion2 = Regex.IsMatch(currentPhone, @"^\+7 \d{3} \d{3}-\d{2}-\d{2}$");

                if (criterion1 && criterion2)
                    lblResultValue.Text = "Корректный номер телефона";
                else
                    lblResultValue.Text = "Не корректный номер телефона";
            }
            catch (Exception ex)
            {
                MessageBox.Show("Ошибка: " + ex.Message);
            }
        }

        private string ExtractValue(string json)
        {
            Match match = Regex.Match(json, "\"value\"\\s*:\\s*\"([^\"]+)\"");
            if (match.Success)
                return match.Groups[1].Value;

            return "";
        }
    }
}
